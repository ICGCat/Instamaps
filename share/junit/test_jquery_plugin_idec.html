jquery.selectOptionsAjax

Example. 

	$('#test_selectOpAjax').selectOptionsAjax({
		url: '/share/dades/json/municipis.json',
		textOption: 'municipi',
		valueOption: 'municipi',
		onComplete: function(){
			//para poner el chosen
			$(this).chosen();
		}
	}).on('change',function(){
		console.debug($(this).find("option:selected"));
		$(this).find("option:selected").each(function(){
			console.debug($(this).data('attributes'));
		});
		console.info($(this).val());
	});
